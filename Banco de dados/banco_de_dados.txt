CREATE DATABASE dona_encrenca;

USE dona_encrenca;

CREATE TABLE cliente (
	id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    telefone VARCHAR(15) NOT NULL,
    cpf VARCHAR(11) NOT NULL UNIQUE,
    email VARCHAR(50) NOT NULL UNIQUE,
    senha VARCHAR(50) NOT NULL
);

CREATE TABLE profissao (
	id_profissao INT(255) AUTO_INCREMENT PRIMARY KEY,
    nome_profissao VARCHAR(255),
	id_cliente INT,
    CONSTRAINT fk_profiente FOREIGN KEY (id_cliente) REFERENCES cliente (id_cliente)
);

CREATE TABLE localidade (
	id_localidade INT(255) AUTO_INCREMENT PRIMARY KEY,
    nome_localidade VARCHAR(255),
	id_cliente INT,
    CONSTRAINT fk_localiente FOREIGN KEY (id_cliente) REFERENCES cliente (id_cliente)
);

CREATE TABLE perfil (
	id INT,
    path VARCHAR(100),
    data_upload DATETIME,
    descricao TEXT
);
