CREATE TABLE cliente (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    telefone VARCHAR(15),
    cpf VARCHAR(11) NOT NULL,
    descricao TEXT,
    profissao_id INT,
    localidade_id INT,
    FOREIGN KEY (profissao_id) REFERENCES profissao(id),
    FOREIGN KEY (localidade_id) REFERENCES localidade(id)
);
